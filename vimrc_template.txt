" disable vi compatiblity
set nocompatible

"security
set nomodeline

"pathogen.vim
call pathogen#infect()

" set UTF-8 encoding
set enc=utf-8
set fenc=utf-8

" set spellchecking
" For specific files
" au FileType mail,markdown,tex setlocal spell spelllang=en
set spell spelllang=en


set background=dark     " Assume a dark background
filetype plugin on

" use indentation of previous line
set autoindent

" use intelligent indentation for C
set smartindent

" turn off word wrap
set nowrap


autocmd BufRead *.py set smartindent cinwords=if,elif,else,for,while,try,except,finally,def,class
let python_highlight_all=1



" configure tabwidth and insert spaces instead of tabs
set tabstop=4        " tab width is 4 spaces
set shiftwidth=4     " indent also with 4 spaces
set expandtab        " expand tabs to spaces

" turn syntax highlighting on
syntax on
colorscheme molokai

" turn line numbers on
set number
set ruler

set incsearch " find as you type search
set hlsearch " highlight search terms
nnoremap <CR> :noh<CR><CR>
set ignorecase                  " case insensitive search

" highlight matching braces
set showmatch
" intelligent comments
set comments=sl:/*,mb:\ *,elx:\ */

"nobackups
set nobackup
set noswapfile

"set pasteToggle
set pastetoggle=<F2>

"no audio bell
set noeb vb t_vb=

" diable gui menu
set guioptions-=T

" set transparency=15          " Make the window slightly transparent

" set font
set guifont=Inconsolata:h14

" NerdTree {
map <A-Space> :NERDTreeToggle<CR>
let NERDTreeShowHidden=1
" }

" NerdTREE_TABS {
let g:nerdtree_tabs_open_on_gui_startup=1
let g:nerdtree_tabs_open_on_console_startup=1
let g:nerdtree_tabs_open_on_new_tab=1 " Love this!
let g:nerdtree_tabs_autoclose=1
" }

" Buffer explorer {
nmap <leader>b :BufExplorer<CR>
" }

" SuperTab option for context aware completion
let g:SuperTabDefaultCompletionType = "<c-x><c-u><c-p>"
autocmd FileType python let g:SuperTabDefaultCompletionType = "<c-x><c-o>"
